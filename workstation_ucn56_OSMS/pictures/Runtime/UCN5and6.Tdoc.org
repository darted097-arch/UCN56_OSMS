function `void CalculatePT_Data()
{
   float AvgWR ;
   float HiValue, LoValue ;

// [1/2]Calculate Highest and Lowest PZR Pressure based on Average of 4CH WR values

      if(Interface->System.displayedUnits)
         AvgWR = (RCP102A->value+RCP102B->value+RCP102C->value+RCP102D->value)/4.0 ;
      else 
         AvgWR = (RCP102A->value+RCP102B->value+RCP102C->value+RCP102D->value)/4.0 * RCP102A->mfactor + RCP102A->moffset ;

      if (AvgWR >= 52.7)
      {
         HiValue = (RCP102A->value>RCP102B->value) ? RCP102A->value:RCP102B->value ;
         HiValue = (RCP102C->value>HiValue)        ? RCP102C->value:HiValue ;
         HiValue = (RCP102D->value>HiValue)        ? RCP102D->value:HiValue ;
       
         LoValue = (RCP102A->value<RCP102B->value) ? RCP102A->value:RCP102B->value ;
         LoValue = (RCP102C->value<LoValue)        ? RCP102C->value:LoValue ;
         LoValue = (RCP102D->value<LoValue)        ? RCP102D->value:LoValue ;
      }
      else
      {
         HiValue = (RCP103J->value>RCP104K->value)   ? RCP103J->value:RCP104K->value ;
         LoValue = (RCP103J->value<RCP104K->value)   ? RCP103J->value:RCP104K->value ;
      }

      HiPZRPress[PT_DBpointer] = (Interface->System.displayedUnits)? HiValue : (HiValue * RCP102A->mfactor +RCP102A->moffset) ; 
      LoPZRPress[PT_DBpointer] = (Interface->System.displayedUnits)? LoValue : (LoValue * RCP102A->mfactor +RCP102A->moffset) ;

// [3]Select the Lower Temp.  of 2 Cold Legs for Normal P-T Curve

      LoColdLegTemp_Normal[PT_DBpointer] = (RCT115->value<RCT125->value) ? RCT115->value:RCT125->value ;

      if(!Interface->System.displayedUnits) 
         LoColdLegTemp_Normal[PT_DBpointer] = LoColdLegTemp_Normal[PT_DBpointer] * RCT115->mfactor + RCT115->moffset ;

// [4]Select the Highest Temp. of 4 Hot Legs for Emergency P-T Curve

      HiValue = (RCT112HA->value>RCT112HB->value) ? RCT112HA->value:RCT112HB->value ;
      HiValue = (RCT122HA->value>HiValue)         ? RCT122HA->value:HiValue ;
      HiValue = (RCT122HB->value>HiValue)         ? RCT122HB->value:HiValue ;

      HiHotLegTemp_Emergency[PT_DBpointer] = (Interface->System.displayedUnits)? HiValue : (HiValue * RCT112HA->mfactor + RCT112HA->moffset) ;

// [5]Select the Lowest Temp. of 4 Cold Legs for Emergency P-T Curve

      LoValue = (RCT112CA->value<RCT112CB->value) ? RCT112CA->value:RCT112CB->value ;
      LoValue = (RCT122CA->value<LoValue)         ? RCT122CA->value:LoValue ;
      LoValue = (RCT122CB->value<LoValue)         ? RCT122CB->value:LoValue ;

      LoColdLegTemp_Emergency[PT_DBpointer] = (Interface->System.displayedUnits)? LoValue : (LoValue * RCT112CA->mfactor + RCT112CA->moffset);

   return;
}`

function `void ParseAndToupper (char *theString, int numberOfChars)
{
   int i;
//
// do until the max number of chars have been processed
   for  (i=0; i<numberOfChars ; i++) 
   {
//    if lowercase, convert 
      if (*theString >= 'a' && *theString <= 'z')
          *theString -= ' ';
//    move to the next character
      ++theString;
   }
// terminate the string and return it
   *theString = '\\0';
}`
